| `test_parse_large_file` | 大文件性能（>10000 行） |

**示例测试**：

```python
import pytest
from repomind.core.parser import TreeSitterParser

@pytest.fixture
def parser():
    return TreeSitterParser("python")

def test_parse_class_definition(parser, tmp_path):
    """测试类定义解析"""
    source = '''
class UserService:
    """用户服务"""
    def __init__(self, db):
        self.db = db

    def get_user(self, user_id: int) -> User:
        return self.db.query(user_id)
'''
    file = tmp_path / "service.py"
    file.write_text(source)

    result = parser.parse_file(str(file))

    assert len(result.symbols) == 3  # UserService + __init__ + get_user

    cls = result.symbols[0]
    assert cls.name == "UserService"
    assert cls.type == "class"
    assert cls.docstring == "用户服务"
    assert cls.start_line == 2
    assert cls.end_line == 8

    methods = [s for s in result.symbols if s.type == "method"]
    assert len(methods) == 2
    assert methods[0].parent_class_id == cls.id

def test_parse_import_relative(parser, tmp_path):
    """测试相对导入解析"""
    source = '''
from .models import User
from ..utils import helper
from ...core.exceptions import ValidationError
'''
    file = tmp_path / "__init__.py"
    file.write_text(source)

    result = parser.parse_file(str(file))

    assert len(result.imports) == 3
    assert result.imports[0].is_relative is True
    assert result.imports[0].relative_level == 1
    assert result.imports[1].relative_level == 2
    assert result.imports[2].relative_level == 3

def test_parse_syntax_error_tolerance(parser, tmp_path):
    """测试语法错误容错"""
    source = '''
def broken_function(
    # 缺少右括号
    x = 1 + 2
    return x
'''
    file = tmp_path / "broken.py"
    file.write_text(source)

    # 不应抛出异常
    result = parser.parse_file(str(file))
    assert result is not None
```

**示例测试**：

```python
import pytest
from repomind.core.type_inference import ProgressiveTypeInference

@pytest.fixture
def engine():
    return ProgressiveTypeInference()

def test_explicit_type_hint(engine):
    """测试显式类型提示推断"""
    source = '''
def process(user: User) -> Order:
    return Order(user)
'''
    result = engine.infer_function_return_type(source, "process")
    assert result.type == "Order"
    assert result.confidence == 0.95
    assert result.strategy == "explicit_hint"

def test_confidence_ordering(engine):
    """测试置信度级联顺序"""
    # 当多种策略都能推断时，应选择置信度最高的
    source = '''
class Service:
    def process(self, data: dict) -> Result:
        return self.transform(data)
'''
    result = engine.infer_call_return_type(source, "Service.process")
    # 显式提示 > self 推断
    assert result.confidence >= 0.85

def test_duck_typing(engine):
    """测试鸭子类型匹配"""
    source = '''
def process(obj):
    obj.save()
    obj.validate()
'''
    result = engine.infer_parameter_type(source, "obj")
    assert result.strategy == "duck_typing"
    assert result.confidence <= 0.60
```

## 8. RCEGen 评测方案

### 8.1 评测维度

| 维度 | 权重 | 评分标准 |
|------|------|---------|
| Correctness | 40% | 修复补丁是否正确解决问题 |
| Clarity | 30% | RCA 报告是否清晰易懂 |
| Depth | 30% | 根因分析是否深入透彻 |

### 8.2 评测流程

1. 准备 10 个公开的 Verified 样本（含 Stack Trace + 期望修复）
2. 对每个样本运行 RepoMind RCA
3. 使用 GPT-4o 作为裁判，对三个维度打分（1-10 分）
4. 计算加权总分

### 8.3 评测指标

| 指标 | 目标值 |
|------|--------|
| RCA 准确率 | ≥ 70% |
| Correctness 均分 | ≥ 7.0/10 |
| Clarity 均分 | ≥ 7.5/10 |
| Depth 均分 | ≥ 6.5/10 |
| Recall@5 | 优于纯向量检索 |
