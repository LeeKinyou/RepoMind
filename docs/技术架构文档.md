# 技术架构文档（Technical Architecture Document）

## 1. 架构概述

### 1.1 设计原则

RepoMind 的技术架构遵循以下核心设计原则：

| 原则 | 说明 | 实践 |
|------|------|------|
| **Local-first** | 优先本地执行，减少外部依赖 | SQLite + LanceDB 本地存储 |
| **Zero-config** | 零配置开箱即用 | 内置所有依赖，无需额外安装 |
| **Modular** | 高内聚低耦合 | 适配器模式隔离语言特定逻辑 |
| **Progressive** | 渐进式增强 | 类型推断从高置信度到低置信度级联 |
| **Secure by Default** | 默认安全 | 沙箱隔离所有危险操作 |

### 1.2 架构风格

采用 **分层架构 + 管道过滤器** 混合风格：


#### 2.2.1 解析引擎（Parser Engine）

```python
class ParserEngine:
    """代码解析引擎，负责 AST 提取和符号解析"""

    def __init__(self):
        self.tree_sitter = TreeSitterParser()  # 语法解析
        self.jedi_engine = JediEngine()        # 类型推断

    def parse_file(self, file_path: str) -> ParsedFile:
        """解析单个文件"""
        # 1. Tree-sitter 解析 AST
        ast = self.tree_sitter.parse(file_path)

        # 2. 提取符号（类、函数、导入）
        symbols = self.extract_symbols(ast)

        # 3. Jedi 类型推断
        type_info = self.jedi_engine.infer_types(file_path)

        # 4. 合并结果
        return ParsedFile(
            path=file_path,
            ast=ast,
            symbols=symbols,
            types=type_info
        )

    def extract_symbols(self, ast) -> List[Symbol]:
        """从 AST 提取符号"""
        symbols = []
        for node in ast.walk():
            if node.type == 'class_definition':
                symbols.append(self.extract_class(node))
            elif node.type == 'function_definition':
                symbols.append(self.extract_function(node))
            elif node.type == 'import_statement':
                symbols.append(self.extract_import(node))
        return symbols
```

#### 2.2.2 类型推断引擎（Type Inference Engine）

```python
class TypeInferenceEngine:
    """渐进式类型推断引擎"""

    def infer_type(self, node: ASTNode) -> Tuple[str, float]:
        """
        渐进式类型推断

        Returns:
            Tuple[str, float]: (类型名称, 置信度)
        """
        # 策略 1: 显式类型提示 (置信度 0.95)
        if type_hint := self.extract_type_hint(node):
            return type_hint, 0.95

        # 策略 2: Import 映射 (置信度 0.90)
        if import_type := self.resolve_import_type(node):
            return import_type, 0.90

        # 策略 3: self 参数 (置信度 0.85)
        if self_type := self.infer_self_type(node):
            return self_type, 0.85

        # 策略 4: 赋值推断 (置信度 0.70)
        if assign_type := self.infer_from_assignment(node):
            return assign_type, 0.70

        # 策略 5: Jedi 推断 (置信度 0.60)
        if jedi_type := self.jedi_infer(node):
            return jedi_type, 0.60

        # 策略 6: 鸭子类型 (置信度 0.40)
        if duck_type := self.infer_duck_type(node):
            return duck_type, 0.40

        return None, 0.0
```

#### 2.2.3 调用图构建器（Call Graph Builder）

```python
class CallGraphBuilder:
    """调用图构建器"""

    def __init__(self, type_engine: TypeInferenceEngine):
        self.type_engine = type_engine
        self.graph = nx.DiGraph()

    def build(self, parsed_files: List[ParsedFile]) -> nx.DiGraph:
        """构建调用图"""
        # 1. 添加所有符号节点
        for pf in parsed_files:
            for symbol in pf.symbols:
                self.graph.add_node(symbol.id, **symbol.to_dict())

        # 2. 构建调用边
        for pf in parsed_files:
            for call in self.extract_calls(pf):
                source = call.caller
                target = self.resolve_call_target(call)

                if target:
                    confidence = self.calculate_confidence(call, target)
                    self.graph.add_edge(
                        source.id,
                        target.id,
                        confidence=confidence,
                        call_type=call.type
                    )

        return self.graph

    def resolve_call_target(self, call: Call) -> Optional[Symbol]:
        """解析调用目标"""
        # 策略 1: 显式类型调用
        if call.receiver_type:
            return self.find_method(call.receiver_type, call.method_name)

        # 策略 2: self 调用
        if call.is_self_call:
            return self.find_in_class(call.caller_class, call.method_name)

        # 策略 3: 全局函数
        if call.is_direct_call:
            return self.find_function(call.function_name)

        # 策略 4: 鸭子类型
        return self.find_by_duck_typing(call.method_name)
```

### 4.1 渐进式类型推断算法

```python
class ProgressiveTypeInference:
    """
    渐进式类型推断算法

    置信度级联：
    1. 显式类型提示: 0.95
    2. Import 映射: 0.90
    3. self 参数: 0.85
    4. 赋值推断: 0.70
    5. Jedi 推断: 0.60
    6. 鸭子类型: 0.40
    """

    STRATEGIES = [
        ('explicit_hint', 0.95),
        ('import_mapping', 0.90),
        ('self_inference', 0.85),
        ('assignment', 0.70),
        ('jedi', 0.60),
        ('duck_typing', 0.40),
    ]

    def infer(self, node: ASTNode, context: AnalysisContext) -> TypeResult:
        """执行渐进式类型推断"""
        for strategy_name, base_confidence in self.STRATEGIES:
            strategy = self.get_strategy(strategy_name)
            result = strategy.infer(node, context)

            if result.success:
                # 根据上下文调整置信度
                adjusted_confidence = self.adjust_confidence(
                    base_confidence,
                    result,
                    context
                )
                return TypeResult(
                    type=result.type,
                    confidence=adjusted_confidence,
                    strategy=strategy_name,
                    evidence=result.evidence
                )

        return TypeResult(type=None, confidence=0.0, strategy='none')

    def adjust_confidence(self, base: float, result, context) -> float:
        """根据上下文调整置信度"""
        confidence = base

        # 如果有多个证据支持，提高置信度
        if len(result.supporting_evidence) > 1:
            confidence = min(confidence + 0.05, 1.0)

        # 如果类型在项目中频繁出现，提高置信度
        if self.is_common_type(result.type, context.project):
            confidence = min(confidence + 0.03, 1.0)

        # 如果存在冲突证据，降低置信度
        if result.conflicting_evidence:
            confidence = max(confidence - 0.10, 0.0)

        return confidence
```

### 4.2 图拓扑扩展算法

```python
class GraphExpansion:
    """
    图拓扑扩展算法

    从种子节点出发，沿调用/继承/导入边向外扩展 N 跳
    """

    def expand(
        self,
        seed_nodes: List[Symbol],
        graph: nx.DiGraph,
        hops: int = 2,
        max_nodes: int = 50
    ) -> ExpansionResult:
        """
        执行图拓扑扩展

        Args:
            seed_nodes: 种子节点列表
            graph: 调用图
            hops: 扩展跳数
            max_nodes: 最大节点数

        Returns:
            ExpansionResult: 扩展结果，包含节点和边
        """
        visited = set()
        frontier = set(seed_nodes)
        all_nodes = set(seed_nodes)
        all_edges = []

        for hop in range(hops):
            next_frontier = set()

            for node in frontier:
                if node.id in visited:
                    continue
                visited.add(node.id)

                # 获取邻居节点
                neighbors = self.get_neighbors(node, graph)

                for neighbor, edge_data in neighbors:
                    if neighbor.id not in visited:
                        next_frontier.add(neighbor)
                        all_nodes.add(neighbor)
                        all_edges.append((node, neighbor, edge_data))

                        # 检查节点数限制
                        if len(all_nodes) >= max_nodes:
                            return ExpansionResult(
                                nodes=list(all_nodes),
                                edges=all_edges,
                                truncated=True
                            )

            frontier = next_frontier

        return ExpansionResult(
            nodes=list(all_nodes),
            edges=all_edges,
            truncated=False
        )

    def get_neighbors(self, node: Symbol, graph: nx.DiGraph) -> List:
        """获取邻居节点（出边 + 入边）"""
        neighbors = []

        # 出边：调用的方法
        for successor in graph.successors(node.id):
            edge_data = graph.edges[node.id, successor]
            neighbors.append((graph.nodes[successor], edge_data))

        # 入边：被调用的方法
        for predecessor in graph.predecessors(node.id):
            edge_data = graph.edges[predecessor, node.id]
            neighbors.append((graph.nodes[predecessor], edge_data))

        return neighbors
```

### 4.3 混合检索算法

```python
class HybridRetriever:
    """
    混合检索器

    结合 BM25 关键字检索和向量语义检索
    """

    def __init__(self, bm25_index, vector_index, graph: nx.DiGraph):
        self.bm25 = bm25_index
        self.vector = vector_index
        self.graph = graph

    def retrieve(
        self,
        query: str,
        top_k: int = 10,
        expand_hops: int = 2
    ) -> RetrievalResult:
        """
        执行混合检索

        1. BM25 关键字检索
        2. 向量语义检索
        3. 结果融合
        4. 图拓扑扩展
        """
        # 1. BM25 检索
        bm25_results = self.bm25.search(query, top_k=top_k * 2)

        # 2. 向量检索
        vector_results = self.vector.search(query, top_k=top_k * 2)

        # 3. 结果融合（RRF - Reciprocal Rank Fusion）
        fused_results = self.fuse_results(bm25_results, vector_results)

        # 4. 取 Top-K
        top_results = fused_results[:top_k]

        # 5. 图拓扑扩展
        expanded = self.expand_with_graph(top_results, hops=expand_hops)

        return RetrievalResult(
            seed_nodes=top_results,
            expanded_nodes=expanded.nodes,
            expanded_edges=expanded.edges
        )

    def fuse_results(self, bm25_results, vector_results, k=60):
        """RRF 结果融合"""
        scores = {}

        for rank, result in enumerate(bm25_results):
            doc_id = result.id
            scores[doc_id] = scores.get(doc_id, 0) + 1 / (k + rank + 1)

        for rank, result in enumerate(vector_results):
            doc_id = result.id
            scores[doc_id] = scores.get(doc_id, 0) + 1 / (k + rank + 1)

        # 按分数排序
        sorted_ids = sorted(scores.keys(), key=lambda x: scores[x], reverse=True)
        return [self.get_result_by_id(id) for id in sorted_ids]
```

## 7. 安全架构

### 7.1 沙箱隔离设计

```python
class SandboxExecutor:
    """
    沙箱执行器

    所有危险操作都在隔离环境中执行
    """

    def __init__(self, use_docker: bool = True):
        self.use_docker = use_docker

    def execute_test(self, test_code: str) -> TestResult:
        """在沙箱中执行测试"""
        if self.use_docker:
            return self.execute_in_docker(test_code)
        else:
            return self.execute_in_subprocess(test_code)

    def execute_in_docker(self, test_code: str) -> TestResult:
        """Docker 沙箱执行"""
        import docker

        client = docker.from_env()

        # 创建临时容器
        container = client.containers.run(
            "repomind-sandbox:latest",
            command=f"pytest {test_code}",
            volumes={
                '/tmp/repomind': {'bind': '/workspace', 'mode': 'ro'}
            },
            mem_limit="512m",
            cpu_period=100000,
            cpu_quota=50000,  # 50% CPU
            network_disabled=True,  # 禁用网络
            remove=True
        )

        return TestResult(
            passed=container.exit_code == 0,
            output=container.logs().decode(),
            exit_code=container.exit_code
        )

    def apply_patch(self, patch: str) -> PatchResult:
        """在沙箱中应用补丁"""
        # 创建临时工作目录
        with tempfile.TemporaryDirectory() as tmpdir:
            # 复制代码
            shutil.copytree(self.project_path, tmpdir)

            # 应用补丁
            result = subprocess.run(
                ["git", "apply", "--check", "-"],
                input=patch,
                capture_output=True,
                cwd=tmpdir
            )

            if result.returncode == 0:
                # 实际应用
                subprocess.run(
                    ["git", "apply", "-"],
                    input=patch,
                    cwd=tmpdir
                )
                return PatchResult(success=True, path=tmpdir)

            return PatchResult(
                success=False,
                error=result.stderr.decode()
            )
```

## 8. 性能优化

### 8.1 并发处理

```python
import asyncio
from concurrent.futures import ThreadPoolExecutor, ProcessPoolExecutor

class ParallelProcessor:
    """并行处理器"""

    def __init__(self, max_workers: int = 4):
        self.thread_pool = ThreadPoolExecutor(max_workers=max_workers)
        self.process_pool = ProcessPoolExecutor(max_workers=max_workers)

    async def parse_files_parallel(
        self,
        files: List[str]
    ) -> List[ParsedFile]:
        """并行解析文件"""
        loop = asyncio.get_event_loop()

        # CPU 密集型任务使用进程池
        tasks = [
            loop.run_in_executor(self.process_pool, self.parse_file, f)
            for f in files
        ]

        return await asyncio.gather(*tasks)

    async def embed_parallel(
        self,
        texts: List[str]
    ) -> List[List[float]]:
        """并行生成嵌入"""
        loop = asyncio.get_event_loop()

        # IO 密集型任务使用线程池
        tasks = [
            loop.run_in_executor(self.thread_pool, self.embed, t)
            for t in texts
        ]

        return await asyncio.gather(*tasks)
```

### 8.2 缓存策略

```python
from functools import lru_cache
import hashlib

class CacheManager:
    """缓存管理器"""

    def __init__(self, cache_dir: str = ".repomind/cache"):
        self.cache_dir = cache_dir
        self.memory_cache = {}

    def get_or_compute(
        self,
        key: str,
        compute_fn: Callable,
        ttl: int = 3600
    ) -> Any:
        """获取缓存或计算"""
        # 1. 内存缓存
        if key in self.memory_cache:
            return self.memory_cache[key]

        # 2. 磁盘缓存
        cache_path = self.get_cache_path(key)
        if os.path.exists(cache_path):
            with open(cache_path, 'rb') as f:
                result = pickle.load(f)
                self.memory_cache[key] = result
                return result

        # 3. 计算
        result = compute_fn()

        # 4. 存储
        self.memory_cache[key] = result
        with open(cache_path, 'wb') as f:
            pickle.dump(result, f)

        return result

    def get_cache_path(self, key: str) -> str:
        """获取缓存文件路径"""
        hash_key = hashlib.md5(key.encode()).hexdigest()
        return os.path.join(self.cache_dir, f"{hash_key}.cache")
```
